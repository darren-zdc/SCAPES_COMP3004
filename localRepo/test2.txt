# This program will loop until b is greater than a
# It also prints the greater number on each loop
# Declare variable a and b
dci a
dci b
# Ask for value of a and b  <--- Loop starts here
L1: rdi a
rdi b
# Compare a and b
cmp a b
# If a > b, goes to L2
jmr L2
# Else: b < a, goes to L3
jmp L3
# L2: a > b
# Prints a and jump back to the loop
L2: prt a
jmp L1
L3: prt b
# Prints b and end the program
end
